def select_numeric_values(data, column):
    df = data.copy()
    df['float_' + column] = data[column].apply(lambda x: (re.sub('[^0-9\.]', '', x.replace(',', '.'))))
    df['float_' + column] = df['float_' + column].replace('', np.NaN).astype(float)
    return df
